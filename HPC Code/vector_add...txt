#include <iostream>
#include <vector>
#include <omp.h>
using namespace std;

int main() {
    int n;
    cout << "Enter size of vectors: ";
    cin >> n;

    vector<int> A(n), B(n), C(n);

    cout << "Enter elements of A:\n";
    for (int i = 0; i < n; i++)
        cin >> A[i];

    cout << "Enter elements of B:\n";
    for (int i = 0; i < n; i++)
        cin >> B[i];

    // Parallel addition using OpenMP
    #pragma omp parallel for
    for (int i = 0; i < n; i++) {
        C[i] = A[i] + B[i];
    }

    cout << "\nResultant vector:\n";
    for (int i = 0; i < n; i++)
        cout << C[i] << " ";

    cout << endl;
    return 0;
}
